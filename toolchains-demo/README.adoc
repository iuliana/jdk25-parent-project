== Toolchain Demo

[source, bash]
----
cd /Users/iulianacosmina/sandbox/jdk25-parent-project
mvn clean install
cd toolchains-demo/target
jar xf toolchains-demo-2.0-SNAPSHOT.jar
cat META-INF/MANIFEST.MF

# expect Java SE 8 = 52 (0x34 hex)
javap -v classes/org/mytoys/Main.class | grep "major version"
----
